CREATE TABLE restaurants(
id INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
name VARCHAR( 40 ) NOT NULL ,
address VARCHAR( 40 ) NOT NULL ,
zip INT ,
city CHAR( 30 ) ,
phone INT ,
website VARCHAR( 50 ) ,
email VARCHAR( 30 ) ,
img VARCHAR( 50 ) ,
lat INT NOT NULL ,
longt INT NOT NULL ,
rating FLOAT,
description VARCHAR( 100 )
)
